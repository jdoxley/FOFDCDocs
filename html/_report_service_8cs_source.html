20 public async Task<ResultBase> CreateReportAsync(int submitter, TrooperReport report)
21 {
22 await using var _dbContext = _dbContextFactory.CreateDbContext();
23 var actual = await _dbContext.FindAsync<Trooper>(submitter);
24 if (actual is null)
25 return new(false, new List<string>() { $"Unable to find a trooper for {submitter}" });
26
27 report.LastUpdate = DateTime.UtcNow;
28 report.SubmittedOn = report.LastUpdate;
29
30 // Get the company for this report.
31 report.ReportViewableAt = (Slot)((int)actual.Slot / 100 * 100);
32
33 // Modify some special cases.
34 switch (report.ReportViewableAt)
35 {
36 case Slot.AcklayCompany:
37 report.ReportViewableAt = Slot.AcklayOne;
38 break;
39 case Slot.InactiveReserve:
40 report.ReportViewableAt = Slot.Hailstorm;
41 break;
42 }
43
44 switch (actual.Slot)
45 {
46 case Slot.AcklayReserve:
47 report.ReportViewableAt = Slot.AcklayOne;
48 break;
49 case Slot.RazorReserve:
50 report.ReportViewableAt = Slot.Razor;
51 break;
52 case Slot.MynockReserve:
53 report.ReportViewableAt = Slot.Mynock;
54 break;
55 }
56
57 // Move company reports to battalion.
58 if (actual.Slot == report.ReportViewableAt)
59 report.ReportViewableAt = Slot.Hailstorm;
60
61 // Send inactive reserves and other like reports up to hailstorm.
62 if (report.ReportViewableAt >= Slot.InactiveReserve)
63 report.ReportViewableAt = Slot.Hailstorm;
64
65 actual.FiledReports.Add(report);
66
67 var tracker = new ReportNotificationTracker()
68 {
69 LastView = report.LastUpdate
70 };
71
72 report.NotificationTrackers.Add(tracker);
73 actual.TrooperReportTrackers.Add(tracker);
74
75 await _dbContext.SaveChangesAsync();
76
77 return new(true, null);
78 }

251 public async Task<TrooperReport?> GetTrooperReportIfAuthorized(Guid report, int viewer, bool manager = false)
252 {
253 await using var _dbContext = _dbContextFactory.CreateDbContext();
254 var actual = await _dbContext.FindAsync<TrooperReport>(report);
255 var viewerActual = await _dbContext.FindAsync<Trooper>(viewer);
256 var viewerClaims = await _dbContext.UserClaims
257 .Where(x => x.UserId == viewer)
258 .ToListAsync();
259
260 if (actual is null || viewerActual is null) return null;
261 // If they are a maanager, pass the report up.
262 if (manager) return actual;
263 // It's their report.
264 if (actual.ReportedById == viewer) return actual;
265 // If they are MP, pass the report up.
266 if (viewerActual.MilitaryPolice)
267 return actual;
268 // if they are battalion, and this report is at battalion,
269 // pass the report up.
270 if (actual.ElevatedToBattalion && viewerActual.Slot == Slot.Hailstorm)
271 return actual;
272 // If they are in the slot that is viewable by this report,
273 // pass the report up.
274 if (actual.ReportViewableAt == viewerActual.Slot)
275 return actual;
276 // If they have a claim to view this report level,
277 // pass the report up.
278 if (viewerClaims.Where(x => x.ClaimType == "Trooper.Report")
279 .Any(x => x.ClaimValue == ((int)actual.ReportViewableAt).ToString()))
280 return actual;
281 // Otherwise return null.
282 return null;
283 }
