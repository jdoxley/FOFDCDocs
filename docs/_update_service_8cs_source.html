277 private async Task<ResultBase> RevertClaimUpdateAsync(Trooper manager, ClaimUpdate update)
278 {
279 if (update.AutomaticChange)
280 return new(false, new()
281 {
282 "Unable to revert claim updates on automatic changes. " +
283 "If needed, remove these manually or revert the change that changed these claims."
284 });
285
286 List<string> errors = new();
287
288 var user = update.ChangedFor;
289
290 List<Claim> toRemove = new();
291 foreach (var data in update.Additions)
292 {
293 toRemove.Add(new(data.Key, data.Value));
294 }
295
296 var identResult = await _userManager.RemoveClaimsAsync(user, toRemove);
297 if (!identResult.Succeeded)
298 {
299 foreach (var err in identResult.Errors)
300 errors.Add($"[{err.Code}] {err.Description}");
301
302 return new(false, errors);
303 }
304
305 List<Claim> toAdd = new();
306 foreach (var data in update.Removals)
307 {
308 toAdd.Add(new(data.Key, data.Value));
309 }
310
311 identResult = await _userManager.AddClaimsAsync(user, toAdd);
312 if (!identResult.Succeeded)
313 {
314 foreach (var err in identResult.Errors)
315 errors.Add($"[{err.Code}] {err.Description}");
316
317 return new(false, errors);
318 }
319
320 user.ClaimUpdates.Add(new()
321 {
322 Additions = update.Removals,
323 Removals = update.Additions,
324 RevertChange = true,
325 ChangedById = manager.Id,
326 ChangedOn = DateTime.UtcNow.ToEst(),
327 });
328
329 identResult = await _userManager.UpdateAsync(user);
330 if (!identResult.Succeeded)
331 {
332 foreach (var err in identResult.Errors)
333 errors.Add($"[{err.Code}] {err.Description}");
334
335 return new(false, errors);
336 }
337
338 return new(true);
339 }
