350 private bool CompareCShopClaims(Dictionary<string, List<string>>? c1,
351 Dictionary<string, List<string>>? c2)
352 {
353 if (c1 is null & c2 is null) return true;
354 if (c1 is null || c2 is null) return false;
355 if (c1.Count != c2.Count) return false;
356 foreach (var i in c1)
357 {
358 if (c2.TryGetValue(i.Key, out var c2Val))
359 {
360 if (!i.Value.SequenceEqual(c2Val)) return false;
361 }
362 else return false;
363 }
364
365 return true;
366 }
