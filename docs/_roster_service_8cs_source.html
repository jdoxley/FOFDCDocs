157 public async Task<RegisterTrooperResult> RegisterTrooper(NewTrooperData trooperData, ClaimsPrincipal user)
158 {
159 List<string> errors = new();
160 try
161 {
162 var uniqueSets = await GetInUseUserDataAsync();
163 if (uniqueSets.Item1.Contains(trooperData.Id))
164 {
165 errors.Add("This ID is already in use.");
166 }
167
168 if (uniqueSets.Item2.Contains(trooperData.NickName))
169 {
170 errors.Add("This Nickname is already in use.");
171 }
172
173 if (errors.Count > 0)
174 return new(false, null, errors);
175
176 var token = Guid.NewGuid().ToString();
177
178 var time = DateTime.UtcNow.ToEst();
179
180 var trooper = new Trooper()
181 {
182 BirthNumber = trooperData.Id,
183 NickName = trooperData.NickName,
184 Rank = trooperData.StartingRank,
185 UserName = token,
186 StartOfService = time,
187 LastPromotion = time,
188 AccessCode = token,
189 Slot = Slot.InactiveReserve,
190 };
191
192 var recruiter = await GetTrooperFromClaimsPrincipalAsync(user);
193 trooper.RecruitedByData = new()
194 {
195 RecruitedById = recruiter?.Id ?? 0,
196 ChangedOn = time,
197 };
198
199 trooper.RecruitStatus = new()
200 {
201 Age = trooperData.Age,
202 ModsInstalled = trooperData.ModsDownloaded,
203 OverSixteen = trooperData.Sixteen,
204 PossibleTroll = trooperData.PossibleTroll,
205 PreferredRole = trooperData.PreferredRole
206 };
207
208 var identRes = await _userManager.CreateAsync(trooper, token);
209
210 if (!identRes.Succeeded)
211 {
212 foreach (var error in identRes.Errors)
213 errors.Add($"[{error.Code}] {error.Description}");
214
215 return new(false, null, errors);
216 }
217
218 identRes = await _userManager.AddToRoleAsync(trooper, "Trooper");
219
220 if (!identRes.Succeeded)
221 {
222 foreach (var error in identRes.Errors)
223 errors.Add($"[{error.Code}] {error.Description}");
224
225 return new(false, null, errors);
226 }
227
228 identRes = await _userManager.AddClaimAsync(trooper, new("Training", "BCT"));
229
230 if (!identRes.Succeeded)
231 {
232 foreach (var error in identRes.Errors)
233 errors.Add($"[{error.Code}] {error.Description}");
234
235 return new(false, null, errors);
236 }
237
238 return new(true, token, null);
239 }
240 catch (Exception ex)
241 {
242 errors.Add(ex.Message);
243 return new(false, null, errors);
244 }
245 }
