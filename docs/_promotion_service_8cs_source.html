100 public async Task<ResultBase> FinalizePromotionAsync(Promotion promotion, Trooper approver)
101 {
102 using var _dbContext = _dbContextFactory.CreateDbContext();
103 var actual = await _dbContext.Promotions.FindAsync(promotion.Id);
104
105 if (actual is null)

108 }
109
110 await _dbContext.Entry(actual).Collection(e => e.ApprovedBy).LoadAsync();
111 await _dbContext.Entry(actual).Reference(e => e.PromotionFor).LoadAsync();
112
113 var user = await _dbContext.FindAsync<Trooper>(approver.Id);
114
115 if (user is null)
116 {
117 return new(false, new() { "Failed to find an account for the approver" });
118 }
119
120 var now = DateTime.UtcNow.ToEst();
121 var log = new RankUpdate()
122 {
123 Approved = true,
124 RequestedById = actual.RequestedById,
125 ChangedFrom = actual.PromoteFrom,
126 ChangedTo = actual.PromoteTo,
127 ChangedForId = actual.PromotionForId,
128 ChangedOn = now,
129 SubmittedByRosterClerk = false,
130 ApprovedBy = actual.ApprovedBy
131 };
132
133 user.ApprovedRankUpdates.Add(log);
134
135 var rank = actual.PromoteTo.GetRank();
136 switch (rank)
137 {
138 case TrooperRank r:
139 actual.PromotionFor.Rank = r;
140 actual.PromotionFor.LastPromotion = now;
141 break;
142 case MedicRank r:
143 actual.PromotionFor.MedicRank = r;
144 break;
145 case RTORank r:
146 actual.PromotionFor.RTORank = r;
147 break;
148 case PilotRank r:
149 actual.PromotionFor.PilotRank = r;
150 actual.PromotionFor.LastPromotion = now;
151 break;
152 case WardenRank r:
153 actual.PromotionFor.WardenRank = r;
154 actual.PromotionFor.LastPromotion = now;
155 break;
156 case WarrantRank r:
157 actual.PromotionFor.WarrantRank = r;
158 actual.PromotionFor.LastPromotion = now;
159 break;
160 }
161
162 _dbContext.Remove(actual);
163
164 await _dbContext.SaveChangesAsync();
165

169 public async Task<PromotionResult> StartPromotionProcessAsync(ClaimsPrincipal invoker, Trooper promotionFor,
170 PromotionBoardLevel currentBoard,
171 int promotionFrom, int promotionTo, string reason, bool forced)
172 {
173 var invoked = await _userManager.GetUserAsync(invoker);
174 var trooper = await _userManager.FindByIdAsync(promotionFor.Id.ToString());
175
176 var promoReq = await _webSettings.GetPromotionRequirementsAsync(promotionTo);
177
178 if (promoReq is null)
179 {
180 return new(false, null, new() { "No promotion board data for the rank this trooper is being promoted to." });
181 }
182
183 // Put this on the highest needed board in the event this person is
184 // eligible.
185 PromotionBoardLevel neededLevel;
186 if (currentBoard > promoReq.NeededLevel)
187 neededLevel = currentBoard;
188 else neededLevel = promoReq.NeededLevel;
189
190 var promotion = new Promotion()
191 {
192 CurrentBoard = currentBoard,
193 NeededBoard = neededLevel,
194
195 PromoteFrom = promotionFrom,
196 PromoteTo = promotionTo,
197 Reason = reason,
198
199 RequestedById = invoked.Id,
200 Forced = forced
201 };
202
203 trooper.PendingPromotions.Add(promotion);
204
205 var identRes = await _userManager.UpdateAsync(trooper);
206 if (!identRes.Succeeded)
207 {
208 List<string> errors = new();
209 foreach (var error in identRes.Errors)
210 errors.Add($"[{error.Code}] {error.Description}");
211
212 return new(false, null, errors);
213 }
214
215 using var _dbContext = _dbContextFactory.CreateDbContext();
216
217 var promo = await _dbContext.Promotions.FindAsync(promotion.Id);
218
219 if (promo is null) return new(false, null, new() { "No promotion object was abled to be retrieved." });
220
221 await _dbContext.Entry(promo).Reference(e => e.PromotionFor).LoadAsync();
222 await _dbContext.Entry(promo).Reference(e => e.RequestedBy).LoadAsync();
223
224 return new(true, promo, null);
225 }
